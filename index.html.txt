    <label>เดือนที่ต้องการ:</label>
    <select id="month">
        <option value="มกราคม">มกราคม</option>
        <option value="กุมภาพันธ์">กุมภาพันธ์</option>
        <option value="มีนาคม">มีนาคม</option>
        </select>

    <label>เวลาที่สะดวก:</label>
    <input type="time" id="time">

    <button onclick="sendData()">ยืนยันข้อมูล</button>

    <script>
        // เปลี่ยน 'YOUR_LIFF_ID' เป็นไอดีที่คุณได้จาก LINE Developers
        async function main() {
            await liff.init({ liffId: "2009331879-7xYXmTYm" });
        }
        main();

        function sendData() {
            const m = document.getElementById('month').value;
            const t = document.getElementById('time').value;
            const msg = `[จองคิว] เดือน: ${m} เวลา: ${t}`;

            liff.sendMessages([{
                type: 'text',
                text: msg
            }]).then(() => {
                liff.closeWindow(); // ส่งเสร็จปิดหน้าต่างทันที
            }).catch((err) => {
                console.error('Error sending message', err);
            });
        }
    </script>
